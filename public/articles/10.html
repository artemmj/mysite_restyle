def binary_search(sorted_list, need_item):
    low_ptr = int(0)                 # –ª–µ–≤—ã–π —É–∫–∞–∑–∞—Ç–µ–ª—å –ø–æ–∏—Å–∫–∞
    high_ptr = len(sorted_list) - 1  # –ø—Ä–∞–≤—ã–π —É–∫–∞–∑–∞—Ç–µ–ª—å –ø–æ–∏—Å–∫–∞

    while low_ptr <= high_ptr:
        # –ü–æ–∫–∞ –Ω–µ –Ω–∞—à–ª–∏, –±–µ—Ä–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —É–∫–∞–∑–∞—Ç–µ–ª—è —Å–µ—Ä–µ–¥–∏–Ω—ã —Å–ø–∏—Å–∫–∞
        mid_ptr = int((low_ptr + high_ptr) / 2)
        check_item = sorted_list[mid_ptr]

        # –ï—Å–ª–∏ —Å–æ–≤–ø–∞–ª —Å —Ç–µ–º –∫–æ—Ç–æ—Ä—ã–π –∏—â–µ–º - –≤–µ—Ä–Ω—É—Ç—å –µ–≥–æ –∏–Ω–¥–µ–∫—Å, –ø—Ä–æ—Ñ–∏—Ç
        if check_item == need_item:
            return mid_ptr
        # –ï—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ –±–æ–ª—å—à–µ, —á–µ–º –∏—â–µ–º - —Å–¥–≤–∏–Ω—É—Ç—å –ø—Ä–∞–≤—É—é –≥—Ä–∞–Ω–∏—Ü—É –ø–æ–∏—Å–∫–∞ –Ω–∞ —Å–µ—Ä–µ–¥–∏–Ω—É - 1
        if check_item > need_item:
            high_ptr = mid_ptr - 1
        # –ï—Å–ª–∏ –º–µ–Ω—å—à–µ - —Å–¥–≤–∏–Ω—É—Ç—å –ª–µ–≤—É—é –≥—Ä–∞–Ω–∏—Ü—É –ø–æ–∏—Å–∫–∞ –Ω–∞ —Å–µ—Ä–µ–¥–∏–Ω—É + 1
        else:
            low_ptr = mid_ptr + 1

    return None

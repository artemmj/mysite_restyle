states = ["mt", "wa", "or", "id", "nv", "ut", "—Å–∞", "az"]

final_stations = set()       # –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –∏—Ç–æ–≥–æ–≤–æ–≥–æ –Ω–∞–±–æ—Ä–∞ —Å—Ç–∞–Ω—Ü–∏–π

stations = {}
# –°–ø–∏—Å–æ–∫ —Å—Ç–∞–Ω—Ü–∏–π, –¥–ª—è –∫–æ—Ç–æ—Ä—ã—Ö —Ç—Ä–µ–±—É–µ—Ç—Å—è –≤—ã–±—Ä–∞—Ç—å –ø–æ–∫—Ä—ã—Ç–∏–µ
# (–∫–ª—é—á - –Ω–∞–∑–≤–∞–Ω–∏–µ, –∑–Ω–∞—á–µ–Ω–∏—è - —à—Ç–∞—Ç—ã –¥–ª–¥—è –∫–æ—Ç–æ—Ä—ã—Ö –≤–µ—â–∞–µ—Ç)
stations["kone"] = set(["id", "nv", "ut"])
stations["ktwo"] = set(["wa", "id", "mt"])
stations["kthree"] = set(["or", "nv", "—Å–∞"])
stations["kfour"] = set(["nv", "ut"])
stations["kfive"] = set(["ca", "az"])

while states_needed:

    # –í—Å–µ —à—Ç–∞—Ç—ã, –æ–±—Å–ª—É–∂–∏–≤–∞–µ–º—ã–µ —ç—Ç–æ–π —Å—Ç–∞–Ω—Ü–∏–µ–π,
    # –∫–æ—Ç–æ—Ä—ã–µ –µ—â–µ –Ω–µ –≤—Ö–æ–¥—è—Ç –≤ —Ç–µ–∫—É—â–µ–µ –ø–æ–∫—Ä—ã—Ç–∏–µ
    states_covered = set()

    for station, states_for_station in stations.items():
        covered = states_needed & states_for_station  # –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤

            states_covered = covered
    states_needed -= states_covered
    final_stations.add(best_station)
